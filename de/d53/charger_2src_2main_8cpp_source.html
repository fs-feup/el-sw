261void chargerMachine() {
262 switch (CH_Status) {
263 case idle: {
264 if (shutdownStatus) {
265 NX_CH_Status = shutdown;
266 } else if (request == 1 and latchingStatus == 1 ) {
267 NX_CH_Status = charging;
268 }
269 break;
270 }
271 case charging: {
272 if (shutdownStatus) {
273 NX_CH_Status = shutdown;
274 } else if (request == 0 or latchingStatus == 0) {
275 NX_CH_Status = idle;
276 }
277 break;
278 }
279 case shutdown:
280 break;
281
282 default: {
283 Serial.println("invalid charger state");
284 break;
