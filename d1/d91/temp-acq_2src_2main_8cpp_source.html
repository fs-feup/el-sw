80void readRawADCData() {
81 for (int adc = 0; adc < N_ADCs; adc++) {
82 for (int channel = 0; channel < N_ADC_CHANNELS; channel++) {
83 if (adc == 7 && channel > 3)
84 continue;
85 int tempID = adc * 8 + channel;
86 switch (tempID) {
87 case 18:
88 case 29:
89 case 33:
90 ADCRaw[adc][channel] = ADCRaw[1][1];
91 break;
92 default:
93 ADCRaw[adc][channel] = ADCs[adc].readADC(channel);
94 break;
95 }
96 Temps[adc][channel] = ADCconversion(ADCRaw[adc][channel]);
97
98 minTemp = min(minTemp, Temps[adc][channel]);
99 maxTemp = max(maxTemp, Temps[adc][channel]);
100 tempSum += Temps[adc][channel];
101 }
102 }
103 avgTemp = tempSum / (N_ADCs * N_ADC_CHANNELS - 4); // Only 60 ADC Channels are usable
104}
